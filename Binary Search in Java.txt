// BINARY SEARCH TEMPLATES FOR JAVA

static int findEqual(int[] arr, int key)
    {
        int low=0, high=arr.length-1;

        while(low<=high)
        {
            int mid=low+(high-low)/2;

            if(arr[mid] > key)
                high=mid-1;
            else if(arr[mid] < key)
                low=mid+1;
            else
                return mid;
        }
        return -1; // not exist
    }

static int findEqualOrJustGreater(int[] arr, int key)
    {
        int low=0, high=arr.length-1;

        int idx=-1;
        while(low<=high)
        {
            int mid=low+(high-low)/2;

            if(arr[mid] >= key)
            {
                idx=mid;
                high=mid-1;
            }
            else
            {
                low=mid+1;
            }
        }
        // returns -1 if no ele greater than or equal exists
        return idx;
    }

static int findEqualOrJustSmaller(int[] arr, int key)

                idx=mid;
                low=mid+1;
            }
            else
            {
                high=mid-1;
            }
        }
        // returns -1 if no ele greater than or equal exists
        return idx;
    }
